void partition(int *arr, int i, int j){ // i | pivot
	int pivot = j;
	int low = i;
	for(int i = low; i < pivot; ++i){
		if(arr[i] < arr[pivot]){
			swap(arr, i, low);
			low++;
		}
	}
	swap(arr, pivot, low);
	return low;
}

void quick_sort(int *arr, int j, int k){
	if(j >= k) return
	int i = partition(arr, j, k);
	quick_sort(arr, j, i-1);
	quick_sort(arr, i+1, k);
}
